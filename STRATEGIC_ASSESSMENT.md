### Trinity Matching Algorithm (`lib/trinity-matching.ts`)

```typescript
interface TrinityProfile {
  quest: string          // What are you becoming?
  service: string        // What do you contribute?
  pledge: string         // What commitments?
  questDepth: number     // 0-100 (how thoroughly explored)
}

interface TrinityScore {
  questScore: number     // 0-100 (job requires this quest)
  serviceScore: number   // 0-100 (job uses this service)
  pledgeScore: number    // 0-100 (job fits this commitment)
  overallScore: number   // Weighted average
}

function calculateTrinityScore(
  profile: TrinityProfile,
  job: Job,
  zepGraph?: ZepGraph
): TrinityScore {
  // Step 1: Extract job Trinity from JD + Zep graph
  const jobTrinity = extractJobTrinity(job, zepGraph)

  // Step 2: Calculate semantic similarity (Claude embeddings)
  const questScore = cosineSimilarity(
    embed(profile.quest),
    embed(jobTrinity.quest)
  ) * 100

  // Step 3: Repeat for Service + Pledge
  // ...

  // Step 4: Weight by questDepth (more explored = higher confidence)
  const confidenceWeight = profile.questDepth / 100
  const overallScore = (
    questScore * 0.4 +
    serviceScore * 0.35 +
    pledgeScore * 0.25
  ) * confidenceWeight

  return {
    questScore: Math.round(questScore),
    serviceScore: Math.round(serviceScore),
    pledgeScore: Math.round(pledgeScore),
    overallScore: Math.round(overallScore)
  }
}
```
